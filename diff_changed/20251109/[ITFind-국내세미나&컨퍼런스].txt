    if(targetURL == null || typeof(targetURL) == 'undefined')
        return true;
   
    targetURL = targetURL.trim();
    
    if(targetURL == "" || targetURL == "#")
        return true;

    if(arURL.length <= 1)
        return true;    //파라미터가 없다면...
    

        
    var postForm = "<form  name='global_hidden_form' method='post' action='"+arURL[0]+"'";

        targetWin = targetWin.trim();
        postForm += " target='"+targetWin+"' ";
    }

    }
    
    postParam += "<input type='hidden' name='form.move' value='true'>";
    postForm += postParam + "</form>";
    
    //alert(postForm);

}
